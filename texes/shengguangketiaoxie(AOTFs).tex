    \frametitle{小型固态近红外光谱仪的设计：原理}
    \begin{enumerate}
        \item 特定频率的信号激励
        \item $\mathrm{TeO}_2$ 晶体中产生声波
        \item 形成周期性的折射率，相当于光子晶体或光栅
        \item 波长满足相位条件的光通过
    \end{enumerate}
